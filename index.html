// CÓDIGO PARA TU GOOGLE APPS SCRIPT (EXCEL) function calcularProximoColor(historialArray) { if (!historialArray || historialArray.length === 0) return "---"; var ultimo = historialArray[0]; var penultimo = historialArray[1]; // 1. Contador de Racha Normal (Ej: Rojo, Rojo, Rojo) var rachaNormal = 1; for(var k = 0; k < historialArray.length - 1; k++){ if(historialArray[k] === historialArray[k+1]) { rachaNormal++; } else { break; } } // 2. Contador de Racha Intercalada / Ping-Pong (Ej: Rojo, Negro, Rojo, Negro) var rachaIntercalada = 1; for(var i = 0; i < historialArray.length - 1; i++){ // Verifica que no sean iguales y que no haya ceros interrumpiendo if(historialArray[i] !== historialArray[i+1] && historialArray[i] !== 'Cero' && historialArray[i+1] !== 'Cero') { rachaIntercalada++; } else { break; } } // LÓGICA BASE: Juega al contrario del último var proxima = (ultimo === "Rojo") ? "NEGRO" : "ROJO"; if (ultimo === 'Cero') { proxima = (penultimo === 'Rojo') ? "ROJO" : "NEGRO"; } // REGLA 1: Se sube a la ola normal si salen 3 iguales seguidos if (rachaNormal >= 3) { proxima = ultimo.toUpperCase(); } // REGLA 2 (LA QUE FALTABA): Se sube a la ola intercalada si detecta el Ping-Pong // Si van 4 tiros alternados, espera que el 5to siga el patrón. if (rachaIntercalada >= 4) { // Si el último fue Rojo, el Ping-Pong dicta que el siguiente es Negro proxima = (ultimo === "Rojo") ? "NEGRO" : "ROJO"; } return proxima; }
